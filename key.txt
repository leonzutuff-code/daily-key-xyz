name: Daily Key Update

on:
  schedule:
    - cron: '5 0 * * *'           # Jeden Tag um 00:05 UTC (ca. 01:05 deutsche Zeit)
  workflow_dispatch:              # Ermöglicht manuellen Start zum Testen

jobs:
  update:
    runs-on: ubuntu-latest
    permissions:
      contents: write             # Wichtig: Erlaubt Commits

    steps:
      - name: Repository checkout
        uses: actions/checkout@v4

      - name: Python setup
        uses: actions/setup-python@v5
        with:
          python-version: '3.11'

      - name: Neuen Key generieren und speichern
        run: |
          python - << 'EOF'
          import random
          import string
          length = 24
          chars = string.ascii_letters + string.digits
          new_key = ''.join(random.choice(chars) for _ in range(length))
          with open('key.txt', 'w', encoding='utf-8') as f:
              f.write(new_key + '\n')
          print(f"Neuer Key generiert: {new_key}")
          EOF

      - name: Changes committen und pushen
        run: |
          git config --global user.name "GitHub Action Bot"
          git config --global user.email "action@github.com"
          
          git add key.txt
          
          # Nur committen, wenn sich wirklich was geändert hat
          git diff --staged --quiet && echo "Kein neuer Key nötig" && exit 0
          
          git commit -m "Daily key update $(date +'%Y-%m-%d')"
          git push
